Apps ScriptをスプシのIDに合わせてデプロイしURLを設定すると、保存時に自動で該当行の実績列に書き込まれます。

function doPost(e) { var ss = SpreadsheetApp.openById( '1EtwwDK4dqKRPdhqFDdDYQ9zDH9hJAk3LuK-1UtpoyGw'); var sh = ss.getSheetByName('202601'); var d = JSON.parse(e.postData.contents); var data = sh.getDataRange().getValues(); var parts = d.date.split('/'); var y=parseInt(parts[0]),m=parseInt(parts[1])-1,day=parseInt(parts[2]); var targetRow = -1; for(var i=1;i<data.length;i++){ var cd=data[i][0]; if(cd instanceof Date&&cd.getFullYear()===y &&cd.getMonth()===m&&cd.getDate()===day){ targetRow=i+1; break; } } if(targetRow===-1) return ContentService.createTextOutput("NOT_FOUND"); var s=sh, r=targetRow; if(d.weight) s.getRange(r,12).setValue(d.weight); if(d.steps) s.getRange(r,13).setValue(d.steps); if(d.burned) s.getRange(r,14).setValue(d.burned); if(d.intake) s.getRange(r,15).setValue(d.intake); if(d.breakfast) s.getRange(r,16).setValue(d.breakfast); if(d.calB) s.getRange(r,17).setValue(d.calB); if(d.lunch) s.getRange(r,18).setValue(d.lunch); if(d.calL) s.getRange(r,19).setValue(d.calL); if(d.dinner) s.getRange(r,20).setValue(d.dinner); if(d.calD) s.getRange(r,21).setValue(d.calD); if(d.training) s.getRange(r,22).setValue(d.training); return ContentService.createTextOutput("OK"); } function doGet(e) { var ss = SpreadsheetApp.openById( '1EtwwDK4dqKRPdhqFDdDYQ9zDH9hJAk3LuK-1UtpoyGw'); var sh = ss.getSheetByName('202601'); var data = sh.getDataRange().getValues(); var parts = e.parameter.date.split('/'); var y=parseInt(parts[0]),m=parseInt(parts[1])-1,day=parseInt(parts[2]); for(var i=1;i<data.length;i++){ var cd=data[i][0]; if(cd instanceof Date&&cd.getFullYear()===y &&cd.getMonth()===m&&cd.getDate()===day){ var row=data[i]; return ContentService.createTextOutput(JSON.stringify({ targetWeight:row[2],targetIntake:row[3],targetBurned:row[4], breakfast:row[6],lunch:row[7],dinner:row[8],training:row[9] })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput('{}') .setMimeType(ContentService.MimeType.JSON); }
